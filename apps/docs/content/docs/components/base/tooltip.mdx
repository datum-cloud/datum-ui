import { Tooltip, Button } from '@/components/ui'

## Overview

Tooltip wraps Radix UI's tooltip primitives into a single convenient component. Pass a `message` and wrap any element as the trigger. The tooltip supports positioning, delay customization, and controlled open state.

## Dependencies

This component requires the following packages:

```package-install
@radix-ui/react-tooltip
```

## Usage

<ComponentPreview name="Basic Tooltip">
  <Tooltip message="This is a tooltip">
    <Button type="secondary" theme="outline">Hover me</Button>
  </Tooltip>
</ComponentPreview>

```tsx
import { Tooltip, Button } from '@datum-cloud/datum-ui'

<Tooltip message="This is a tooltip">
  <Button type="secondary" theme="outline">Hover me</Button>
</Tooltip>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `message` | `string \| ReactNode` | -- | Tooltip content |
| `children` | `ReactNode` | -- | Trigger element (rendered with `asChild`) |
| `delayDuration` | `number` | `200` | Delay in ms before tooltip appears |
| `side` | `'top' \| 'right' \| 'bottom' \| 'left'` | -- | Preferred side of the trigger |
| `align` | `'start' \| 'center' \| 'end'` | -- | Alignment along the side |
| `sideOffset` | `number` | `0` | Distance in px from the trigger |
| `hidden` | `boolean` | -- | Hide the tooltip content |
| `open` | `boolean` | -- | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | -- | Callback when open state changes |
| `contentClassName` | `string` | -- | Class for the tooltip content container |
| `arrowClassName` | `string` | -- | Class for the tooltip arrow |
