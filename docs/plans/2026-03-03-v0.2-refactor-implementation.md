# datum-ui v0.2 Refactor Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Refactor datum-ui into a production-grade, plug-and-play design system with tsdown, Rsbuild Storybook, ESLint 10, Vitest, provider-based theming, and per-component sub-path exports.

**Architecture:** Fresh-start migration on `feat/v0.2-refactor` branch. Rebuild monorepo scaffold from scratch with 4 packages (`datum-ui`, `shadcn`, `utils`, `config`) and 2 apps (`storybook`, `docs`). Migrate existing components and styles, add DatumProvider with absorbed datum-themes module.

**Tech Stack:** TypeScript 5.x (isolatedDeclarations), React 19, tsdown, Rsbuild, Storybook 10, Fumadocs, Tailwind CSS v4, CVA, Vitest, ESLint 10 (@antfu/eslint-config), Changesets

**Reference:** See `docs/plans/2026-03-03-v0.2-refactor-design.md` for the full approved design.

**Step 6: Create provider/datum-provider.tsx**

```tsx
'use client'

import '../styles/root.css'
import { ThemeProvider } from './theme-provider'
import type { ReactNode } from 'react'
import type { Theme } from './types'

export interface DatumProviderProps {
  children: ReactNode
  defaultTheme?: Theme
  forcedTheme?: Theme
  storageKey?: string
  disableTransitionOnChange?: boolean
  nonce?: string
}

export function DatumProvider({
  children,
  defaultTheme = 'system',
  storageKey = 'datum-theme',
  ...props
}: DatumProviderProps): React.JSX.Element {
  return (
    <ThemeProvider
      attribute="class"
      defaultTheme={defaultTheme}
      storageKey={storageKey}
      enableSystem
      enableColorScheme
      value={{ light: 'theme-alpha', dark: 'theme-alpha dark' }}
      {...props}
    >
      {children}
    </ThemeProvider>
  )
}
```

```tsx
import { render, screen, act } from '@testing-library/react'
import { describe, it, expect, beforeEach } from 'vitest'
import { DatumProvider } from '../datum-provider'
import { useTheme } from '../theme-provider'

function ThemeDisplay(): React.JSX.Element {
  const { theme, resolvedTheme } = useTheme()
  return (
    <div>
      <span data-testid="theme">{theme}</span>
      <span data-testid="resolved">{resolvedTheme}</span>
    </div>
  )
}

describe('DatumProvider', () => {
  beforeEach(() => {
    localStorage.clear()
    document.documentElement.className = ''
    document.documentElement.removeAttribute('data-theme')
  })

  it('renders children', () => {
    render(
      <DatumProvider>
        <div data-testid="child">Hello</div>
      </DatumProvider>,
    )
    expect(screen.getByTestId('child')).toHaveTextContent('Hello')
  })

  it('defaults to system theme', () => {
    render(
      <DatumProvider>
        <ThemeDisplay />
      </DatumProvider>,
    )
    expect(screen.getByTestId('theme')).toHaveTextContent('system')
  })

  it('respects defaultTheme prop', () => {
    render(
      <DatumProvider defaultTheme="dark">
        <ThemeDisplay />
      </DatumProvider>,
    )
    expect(screen.getByTestId('theme')).toHaveTextContent('dark')
  })

  it('persists theme to localStorage with custom key', () => {
    function ThemeChanger(): React.JSX.Element {
      const { setTheme } = useTheme()
      return <button onClick={() => setTheme('dark')}>Dark</button>
    }

    render(
      <DatumProvider storageKey="datum-theme">
        <ThemeChanger />
      </DatumProvider>,
    )

    act(() => {
      screen.getByText('Dark').click()
    })

    expect(localStorage.getItem('datum-theme')).toBe('dark')
  })
})
```
